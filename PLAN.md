## Decisions Already Made (do not re-debate)

| Decision | Choice | Why |
|---|---|---|
| Data in git? | **No.** Data is regenerated by CI + build. | Free to host, no LFS quota, clean history |
| LFS? | **Only for true binaries** (images, mbtiles, shapefiles, zips) | JSON/CSV go in `.gitignore` now |
| Canonical schema | **Flat-ish with nested `metrics` and `location`** (see Phase 2) | Matches existing transformers; minimal migration |
| Hosting | **Netlify Functions** (free tier) + **GitHub Actions** for daily cache refresh | Zero cost, already configured |
| Data freshness | **Cached, refreshed daily by cron** | Balance: fast deploys + up-to-date |
| Safety | **Create `pre-lfs-cleanup` branch** before destructive changes | Single-command rollback |

## Phase 2 — Schema Reconciliation

**Goal:** One canonical record shape, version 3.0.0, enforced at the validator boundary.

### 2.1 Define canonical schema

Create `scripts/utils/canonical-schema.js` (new file), delete `scripts/utils/standardized-schema.js` at the end:

```js
/**
 * Canonical Unified Schema v3.0.0
 * This is the single source of truth for record shape.
 * Every transformer MUST produce records conforming to this.
 */

export const SCHEMA_VERSION = '3.0.0';

export const CATEGORIES = [
  'conflict', 'martyrs', 'health', 'education', 'economic',
  'humanitarian', 'refugees', 'infrastructure', 'water',
  'culture', 'land', 'historical', 'demographics', 'prisoners'
];

export const REGIONS = ['Gaza Strip', 'West Bank', 'East Jerusalem', 'Palestine'];

export const CONFLICT_PHASES = [
  'nakba-1948', 'naksa-1967', 'first-intifada', 'oslo-period',
  'second-intifada', 'gaza-2008-2009', 'gaza-2012', 'gaza-2014',
  'gaza-2021', 'pre-escalation-2023', 'gaza-war-2023', 'ongoing'
];

/**
 * Canonical record structure (all fields optional except id/date/category).
 *
 * {
 *   id:          string   // unique, format: "{source}-{category}-{hash}"
 *   date:        string   // ISO 8601 YYYY-MM-DD
 *   category:    string   // one of CATEGORIES
 *   event_type:  string   // source-specific event type, lowercase snake_case
 *   schema_version: string // always SCHEMA_VERSION
 *
 *   location: {
 *     name:         string  // human-readable (city, camp, governorate)
 *     governorate:  string  // admin level 1
 *     region:       string  // one of REGIONS
 *     lat:          number  // WGS84
 *     lon:          number  // WGS84
 *     precision:    string  // "exact" | "city" | "governorate" | "region" | "unknown"
 *   },
 *
 *   metrics: {
 *     killed:      number
 *     injured:     number
 *     displaced:   number
 *     affected:    number
 *     demolished:  number
 *     detained:    number
 *     count:       number   // generic count
 *     value:       number   // for indicator-style data (GDP, unemployment %)
 *     unit:        string   // "persons" | "USD" | "percent" | "structures" | ...
 *   },
 *
 *   description:    string
 *   actors:         string[]    // parties involved
 *   severity_index: number      // 0-1 (transformer-computed)
 *
 *   temporal_context: {
 *     days_since_baseline: number
 *     baseline_date:       string  // "2023-10-07"
 *     conflict_phase:      string  // one of CONFLICT_PHASES
 *   },
 *
 *   quality: {
 *     score:        number  // 0-1
 *     completeness: number
 *     consistency:  number
 *     accuracy:     number
 *     confidence:   string  // "high" | "medium" | "low"
 *     verified:     boolean
 *   },
 *
 *   sources: [{
 *     name:         string
 *     organization: string
 *     url:          string
 *     license:      string   // "CC-BY-4.0" | "ODbL" | "public-domain" | ...
 *     fetched_at:   string   // ISO timestamp
 *   }]
 * }
 */

export function createEmptyRecord() {
  return {
    id: null, date: null, category: null, event_type: null,
    schema_version: SCHEMA_VERSION,
    location: { name: null, governorate: null, region: null, lat: null, lon: null, precision: 'unknown' },
    metrics: { killed: 0, injured: 0, displaced: 0, affected: 0, demolished: 0, detained: 0, count: 0, value: 0, unit: null },
    description: '', actors: [], severity_index: 0,
    temporal_context: { days_since_baseline: null, baseline_date: '2023-10-07', conflict_phase: null },
    quality: { score: 0, completeness: 0, consistency: 0, accuracy: 0, confidence: 'low', verified: false },
    sources: []
  };
}

export function validateRecord(record) {
  const errors = [];
  if (!record.id) errors.push('missing id');
  if (!record.date) errors.push('missing date');
  if (!record.category) errors.push('missing category');
  if (record.category && !CATEGORIES.includes(record.category)) {
    errors.push(`invalid category: ${record.category}`);
  }
  if (record.date && isNaN(new Date(record.date).getTime())) {
    errors.push(`invalid date: ${record.date}`);
  }
  if (record.location?.lat != null && (record.location.lat < 29 || record.location.lat > 34)) {
    errors.push(`lat out of Palestine bounds: ${record.location.lat}`);
  }
  if (record.location?.lon != null && (record.location.lon < 33 || record.location.lon > 36)) {
    errors.push(`lon out of Palestine bounds: ${record.location.lon}`);
  }
  return { valid: errors.length === 0, errors };
}
```

```js
toCanonical(partial) {
  const base = createEmptyRecord();

    schema_version: SCHEMA_VERSION,

determineConflictPhase(date) {
  const d = new Date(date);
  if (isNaN(d)) return null;
  const y = d.getUTCFullYear();
  if (y <= 1949) return 'nakba-1948';

  if (d < new Date('2014-09-01')) return 'gaza-2014';
```

```js
import fs from 'fs/promises';
import path from 'path';

const DATA_DIR = path.join(process.cwd(), 'public', 'data', 'unified');

export async function listCategories(req, res) {
  try {
    const entries = await fs.readdir(DATA_DIR, { withFileTypes: true });
    const categories = [];
    for (const e of entries) {
      if (!e.isDirectory()) continue;
      const metaPath = path.join(DATA_DIR, e.name, 'metadata.json');
      try {
        const meta = JSON.parse(await fs.readFile(metaPath, 'utf8'));
        categories.push({ name: e.name, ...meta });
      } catch {
        categories.push({ name: e.name, error: 'no metadata' });
      }
    }
    res.json({ count: categories.length, categories });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}
```

```js
import { getUnifiedData, categoryExists } from '../utils/fileService.js';

export async function summary(req, res) {
  const { category } = req.params;
  if (!await categoryExists(category)) return res.status(404).json({ error: 'not found' });
  const { data } = await getUnifiedData(category);

  const byRegion = {};
  const byYear = {};
  const byEventType = {};
  let totalKilled = 0, totalInjured = 0;

  for (const r of data) {
    const region = r.location?.region || 'Unknown';
    byRegion[region] = (byRegion[region] || 0) + 1;
    const year = r.date?.slice(0, 4) || 'Unknown';
    byYear[year] = (byYear[year] || 0) + 1;
    const et = r.event_type || 'unknown';
    byEventType[et] = (byEventType[et] || 0) + 1;
    totalKilled += r.metrics?.killed || 0;
    totalInjured += r.metrics?.injured || 0;
  }

  res.json({
    category,
    total_records: data.length,
    totals: { killed: totalKilled, injured: totalInjured },
    by_region: byRegion,
    by_year: byYear,
    by_event_type: byEventType
  });
}

export async function timeseries(req, res) {
  const { category } = req.params;
  const { interval = 'month', metric = 'count' } = req.query;
  if (!await categoryExists(category)) return res.status(404).json({ error: 'not found' });
  const { data } = await getUnifiedData(category);

  const buckets = {};
  for (const r of data) {
    if (!r.date) continue;
    let key;
    if (interval === 'year')  key = r.date.slice(0, 4);
    else if (interval === 'month') key = r.date.slice(0, 7);
    else if (interval === 'day')   key = r.date.slice(0, 10);
    else return res.status(400).json({ error: 'interval must be year|month|day' });

    if (!buckets[key]) buckets[key] = { killed: 0, injured: 0, count: 0 };
    buckets[key].killed  += r.metrics?.killed || 0;
    buckets[key].injured += r.metrics?.injured || 0;
    buckets[key].count   += 1;
  }

  const series = Object.entries(buckets)
    .sort(([a], [b]) => a.localeCompare(b))
    .map(([date, values]) => ({ date, ...values }));

  res.json({ category, interval, series });
}
```
